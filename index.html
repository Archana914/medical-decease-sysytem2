from flask import Flask, request, jsonify, render_template_string import joblib # Load the model and vectorizer model = joblib.load("disease_model.joblib") vectorizer = joblib.load("vectorizer.joblib") # Initialize Flask app = Flask(_name_) # Simple HTML template (form) HTML_TEMPLATE = """ Disease Prediction

Disease Prediction System


{% if prediction %}

Predicted Disease: {{ prediction }}

{% endif %} """ @app.route("/", methods=["GET"]) def home(): return render_template_string(HTML_TEMPLATE) @app.route("/predict", methods=["POST"]) def predict(): try: symptoms = request.form["symptoms"] symptoms_vec = vectorizer.transform([symptoms]) prediction = model.predict(symptoms_vec)[0] return render_template_string(HTML_TEMPLATE, prediction=prediction) except Exception as e: return jsonify({"error": str(e)}) if _name_ == "_main_": print("âœ… Flask app started successfully!") app.run(host="127.0.0.1", port=5000, debug=True, use_reloader=False)
